// Test Comment
template TestTemplate
{
    project "$(MODULE_NAME) : project"
    project "$(MODULE_NAME).Tests : project.tests"

    settings project
    {
        set include files: glob "*.{cs,txt,json,xml,md}"
        set exclude files: glob "**/Tests/"

        set target framework: v4.6
        set language version: 6

        add lib refs
        [
            Lib1.dll
            Lib2.dll
        ]

        add define constants
        [
            DEFINE_A
            DEFINE_B
        ]

        if (debug) add define constants
        [
            DEBUG
            TRACE
        ]

        if (release) add define constants
        [
            RELEASE
        ]
        
        if (true)
        {
            if (true)
            {
                set nested block test: true
            }
        }
    }

    settings project.tests : project
    {
        if (no-tests) exclude
        if (!test) skip

        set include files: glob "**/Tests/**/*.{cs,txt,json,xml,md}"
        set exclude files: empty

        add project refs
        [
            $(MODULE_NAME)
        ]
    }
}
