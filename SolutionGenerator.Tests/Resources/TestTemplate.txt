        exclude files = glob "**/Tests/"

        [
            Lib1.dll
            Lib2.dll
        ]

        define constants +=
        [
            DEFINE_A
            DEFINE_B
        ]

        if (debug) define constants +=
        [
            DEBUG
            TRACE
        ]

        if (release) define constants +=
        [
            RELEASE
        ]
        
        if (true)
        {
            if (true)
            {
                nested block test += true
            }
        }
    }

    settings project.tests : project
    {
        if (no-tests) exclude
        if (!test) skip

        include files = glob "**/Tests/**/*.{cs,txt,json,xml,md}"
        exclude files = empty

        project refs +=
        [
            $(MODULE_NAME)
        ]
    }
}
