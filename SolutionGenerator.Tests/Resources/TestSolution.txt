solution MySolution
{
    configuration everything
    [
        Debug: debug,test
        Release: release
    ]

    configuration no-tests
    [
        Debug: debug
        Release: release
    ]

    set target platforms
    [
        Any CPU
    ]

    set root namespace: MySolution
}

module TestModule : TestTemplate
{
}

module OtherTestModule : TestTemplate
{
    settings project
    {
        add project refs
        [
            TestModule
        ]
    }

    settings project.tests : project
    {
        add project refs
        [
            TestModule.Tests
        ]
    }
}

template TestTemplate
{
    add project $(MODULE_NAME) : project
    add project $(MODULE_NAME).Tests : project.tests

    settings project
    {
        set include files: glob "**/$(MODULE_NAME)/**/*.{cs,txt,json,xml,md}"
        set exclude files: glob "**/Tests/**"

        set target framework: v4.6
        set language version: 6

        set debug symbols (debug): true
        set debug type (debug): full
        set optimize (debug): false

        set debug symbols (release): false
        set debug type (release): pdbonly
        set optimize (release): true

        set error report: prompt
        set warning level: 4
        set platform target: AnyCPU

        add lib refs
        [
            Lib1.dll
            Lib2.dll
        ]

        add define constants
        [
            DEFINE_A
            DEFINE_B
        ]

        add define constants (debug)
        [
            DEBUG
            TRACE
        ]

        add define constants (release)
        [
            RELEASE
        ]
    }

    settings project.tests : project
    {
        // Clear the inherited files
        set include files: none
        set exclude files: none

        exclude (no-tests)
        skip (!test)

        set include files: glob "**/Tests/**/*.{cs,txt,json,xml,md}"
        set exclude files: empty

        add project refs
        [
            $(MODULE_NAME)
        ]
    }
}
